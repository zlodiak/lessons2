есть приложение на bottle и требуется собрать контейнер:

FROM python:3.6.8
COPY . .
RUN /bin/bash -c "python3 -m pip install -r requirements.txt"
CMD python3 index.py

сборка:
docker build -t bottler .

запуск:
docker run -i -t --rm -p 5000:5000 bottler
