подобно mergeMap позволяет избежать вложенных запросов, но в отличие от него отправляет в subscribe результаты в порядке поступления первичных запросов

of(1, 2, 4).pipe(
  concatMap(vl => of(vl).pipe(delay(1000 / vl)))
).subscribe(vl => console.log(vl));

// 1, 2, 4
