def bubble_sort(A):
    for bypass in range(1, len(A)):
        for k in range(len(A)-bypass):
            if A[k] > A [k+1]:
                A[k], A[k+1] = A[k+1], A[k]

A = [5, 3, 4, 1, 2]
print(A)
bubble_sort(A)
