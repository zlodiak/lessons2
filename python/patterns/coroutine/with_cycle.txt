def average():
  count = 0
  summ = 0
  average = None
  print('init')

  while True:
    try:
      x = yield average
    except StopIteration:
      print('Done')
    else:     # выполняется если не произошло исключения
      count += 1
      summ += x
      average = round(summ / count, 2)


g = average()         # init пока не выводится
print(g.send(None))   # init выводится, yield возвращает None
